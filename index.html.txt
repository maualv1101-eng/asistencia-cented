    <script>
        // CONFIGURACIÓN CENTRAL
        const CLAVE_MAESTRA = '111010';
        const CARACTERES = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
        
        // Coloca aquí la dirección que obtuviste en el Paso 1.3
        const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbyuxBs-k9c690giwF6qukndaAYkb_P1Y-NdAbXkAPjjhcYM_mmNsJ9t6Ew_c2JHEQpi/exec"; 

        // Mostrar formulario
        function mostrarFormulario(formulario) {
            document.querySelectorAll('.formulario').forEach(el => {
                el.classList.remove('activo');
            });

            if (formulario === 'registro') {
                document.getElementById('formRegistro').classList.add('activo');
            } else if (formulario === 'generarClave') {
                document.getElementById('formGenerarClave').classList.add('activo');
                document.getElementById('claveGenerada').style.display = 'none';
                limpiarFormularioGenerarClave();
            } else if (formulario === 'verRegistros') {
                document.getElementById('formVerRegistros').classList.add('activo');
            }
        }

        // Volver al menú principal
        function volverMenuPrincipal() {
            document.querySelectorAll('.formulario').forEach(el => {
                el.classList.remove('activo');
            });
            document.getElementById('menuPrincipal').classList.add('activo');
            limpiarFormularios();
        }

        // Limpiar formularios
        function limpiarFormularios() {
            document.getElementById('nombreAsistencia').value = '';
            document.getElementById('claveEstudiante').value = '';
            document.getElementById('grupoAsistencia').value = '';
            document.getElementById('claveDocente').value = '';
            document.getElementById('claveDocenteRegistros').value = '';
            document.getElementById('alertRegistro').innerHTML = '';
            document.getElementById('alertGenerarClave').innerHTML = '';
            document.getElementById('alertVerRegistros').innerHTML = '';
        }

        function limpiarFormularioGenerarClave() {
            document.getElementById('nombreNuvoEstudiante').value = '';
            document.getElementById('claveDocenteGenerar').value = '';
            document.getElementById('alertGenerarClave').innerHTML = '';
        }

        // Validar formato de nombre
        function validarNombre(nombre) {
            const regex = /^[A-Z][a-z]+(\s[A-Z][a-z]+)+$/;
            return regex.test(nombre.trim());
        }

        // Normalizar nombre
        function normalizarNombre(nombre) {
            return nombre.trim().split(' ').map(palabra => {
                return palabra.charAt(0).toUpperCase() + palabra.slice(1).toLowerCase();
            }).join(' ');
        }

        // Validar código
        function validarCodigo(codigo) {
            const regex = /^[A-Z0-9]{4}$/;
            return regex.test(codigo.toUpperCase());
        }

        // Mostrar alerta
        function mostrarAlerta(elementId, mensaje, tipo) {
            const elemento = document.getElementById(elementId);
            elemento.innerHTML = `<div class="alert alert-${tipo}">${mensaje}</div>`;
        }

        // ENVIAR ASISTENCIA A GOOGLE SHEETS
        function registrarAsistencia() {
            const nombre = document.getElementById('nombreAsistencia').value;
            const clave = document.getElementById('claveEstudiante').value.toUpperCase();
            const grupo = document.getElementById('grupoAsistencia').value;
            const claveDocente = document.getElementById('claveDocente').value;
            const btn = document.getElementById('btnRegistrar');

            // Validaciones
            if (!nombre || !clave || !grupo || !claveDocente) {
                mostrarAlerta('alertRegistro', '⚠️ Completa todos los campos', 'warning');
                return;
            }

            const nombreNormalizado = normalizarNombre(nombre);

            if (!validarNombre(nombreNormalizado)) {
                mostrarAlerta('alertRegistro', '❌ Nombre inválido. Formato: Nombre Apellido (Primera letra mayúscula)', 'error');
                return;
            }

            if (!validarCodigo(clave)) {
                mostrarAlerta('alertRegistro', '❌ Clave debe tener 4 caracteres (letras mayúsculas y números)', 'error');
                return;
            }

            if (claveDocente !== CLAVE_MAESTRA) {
                mostrarAlerta('alertRegistro', '❌ Clave de docente incorrecta', 'error');
                return;
            }

            // Deshabilitar botón para evitar múltiples envíos
            btn.disabled = true;
            btn.textContent = "⌛ Registrando en la nube...";

            // Preparar los datos
            const formData = new URLSearchParams();
            formData.append('nombre', nombreNormalizado);
            formData.append('clave', clave);
            formData.append('grupo', grupo);

            // Enviar petición POST a Google Apps Script
            fetch(SCRIPT_URL, {
                method: 'POST',
                mode: 'no-cors', // Evita problemas de CORS comunes con Google Sheets
                body: formData,
                headers: {
                    'Content-Type': 'application/x-www-form-urlencoded'
                }
            })
            .then(() => {
                // Como usamos no-cors, asumimos éxito si la petición no falló físicamente
                mostrarAlerta('alertRegistro', `✅ ¡Perfecto! Asistencia registrada para ${nombreNormalizado}`, 'success');
                
                // Limpiar campos
                document.getElementById('nombreAsistencia').value = '';
                document.getElementById('claveEstudiante').value = '';
                document.getElementById('grupoAsistencia').value = '';
                document.getElementById('claveDocente').value = '';
            })
            .catch(error => {
                console.error('Error:', error);
                mostrarAlerta('alertRegistro', '❌ Error de conexión al guardar la asistencia.', 'error');
            })
            .finally(() => {
                btn.disabled = false;
                btn.textContent = "✅ Registrar Asistencia";
            });
        }

        // GENERAR CLAVE
        function generarClave() {
            const nombre = document.getElementById('nombreNuvoEstudiante').value;
            const claveDocente = document.getElementById('claveDocenteGenerar').value;

            if (!nombre || !claveDocente) {
                mostrarAlerta('alertGenerarClave', '⚠️ Completa todos los campos', 'warning');
                return;
            }

            const nombreNormalizado = normalizarNombre(nombre);

            if (!validarNombre(nombreNormalizado)) {
                mostrarAlerta('alertGenerarClave', '❌ Nombre inválido. Formato: Nombre Apellido', 'error');
                return;
            }

            if (claveDocente !== CLAVE_MAESTRA) {
                mostrarAlerta('alertGenerarClave', '❌ Clave de docente incorrecta', 'error');
                return;
            }

            // Generar clave aleatoria simple de 4 dígitos
            let claveNueva = '';
            for (let i = 0; i < 4; i++) {
                claveNueva += CARACTERES.charAt(Math.floor(Math.random() * CARACTERES.length));
            }

            // Mostrar clave generada
            document.getElementById('nomGenerado').textContent = nombreNormalizado;
            document.getElementById('codGenerado').textContent = claveNueva;
            document.getElementById('claveGenerada').style.display = 'block';

            mostrarAlerta('alertGenerarClave', '✅ Clave generada. ¡Recuerda guardarla!', 'success');
        }

        // REDIRECCIÓN A REGISTROS
        function abrirRegistros() {
            const claveDocente = document.getElementById('claveDocenteRegistros').value;

            if (!claveDocente) {
                mostrarAlerta('alertVerRegistros', '⚠️ Ingresa tu clave de docente', 'warning');
                return;
            }

            if (claveDocente !== CLAVE_MAESTRA) {
                mostrarAlerta('alertVerRegistros', '❌ Clave incorrecta', 'error');
                return;
            }

            mostrarAlerta('alertVerRegistros', '✅ Abriendo tu hoja de cálculo...', 'success');
            setTimeout(() => {
                // Abre el archivo del Google Sheet en una pestaña nueva
                alert("🔐 Abre tu Google Drive para visualizar tu hoja de cálculo 'Asistencia Grupo Cented' de manera segura.");
                volverMenuPrincipal();
            }, 1000);
        }
